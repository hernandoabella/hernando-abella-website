  },
  addSound: function() {
    waveSource.setAttribute('src', 'sounds/' + tapAudio.sounds[tapAudio.randomSound()] + '.wav');
    audioPlayer.load();
  },
  play: function() {
    tapAudio.addSound();
    audioPlayer.currentTime = 0;
    audioPlayer.play();
  }
}
